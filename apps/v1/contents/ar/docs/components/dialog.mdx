  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from "@/components/ui/dialog"

export default function Example() {
  return (
    <Dialog>
      <DialogTrigger>فتح</DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>هل أنت متأكد تمامًا؟</DialogTitle>

        </DialogHeader>
      </DialogContent>

  )
}`}
